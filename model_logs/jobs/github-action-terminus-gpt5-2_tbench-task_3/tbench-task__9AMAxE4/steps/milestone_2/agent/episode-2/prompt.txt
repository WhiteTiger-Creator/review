static const char *get_content_type(const char *path) {
    const char *ext = strrchr(path, '.');
    if (!ext) return "application/octet-stream";
    if (strcmp(ext, ".html") == 0) return "text/html";
    if (strcmp(ext, ".css") == 0) return "text/css";
    if (strcmp(ext, ".js") == 0) return "application/javascript";
    return "application/octet-stream";
}

static int url_contains_encoded_dotdot(const char *url) {
    const char *p = url;
    while (*p) {
        if (strncasecmp(p, "%2e%2e", 6) == 0)
            return 1;
        p++;
    }
    return 0;
}

static int from_hex(char c) {
    if (c >= '0' && c <= '9') return c - '0';
    if (c >= 'a' && c <= 'f') return 10 + (c - 'a');
    if (c >= 'A' && c <= 'F') return 10 + (c - 'A');
    return -1;
}

// Percent-decode URL path into out (NUL-terminated). Returns 0 on success.
static int url_percent_decode(const char *in, char *out, size_t out_sz) {
    size_t oi = 0;
    for (size_t i = 0; in[i] != '\0'; i++) {
        if (oi + 1 >= out_sz) return -1;
        if (in[i] == '%' && in[i+1] && in[i+2]) {
            int hi = from_hex(in[i+1]);
            int lo = from_hex(in[i+2]);
            if (hi < 0 || lo < 0) return -1;
            out[oi++] = (char)((hi << 4) | lo);
            i += 2;
            continue;
        }
        out[oi++] = in[i];
    }
    out[oi] = '\0';
    return 0;
}

static int path_has_dot_segments(const char *path) {
    // Reject "." or ".." path segments after decoding.
    const char *p = path;
    while (*p) {
        while (*p == '/') p++;
        if (*p == '\0') break;
        const char *seg = p;
        while (*p && *p != '/') p++;
        size_t len = (size_t)(p - seg);
        if ((len == 1 && seg[0] == '.') ||
            (len == 2 && seg[0] == '.' && seg[1] == '.'))
            return 1;
    }
    return 0;
}

static enum MHD_Result serve_static_file(struct MHD_Connection *connection,

    snprintf(filepath, sizeof(filepath), "%s%s", WWW_DIR, path);

    struct stat st;
    if (stat(filepath, &st) != 0) return MHD_NO;

    MHD_destroy_response(response);
    return ret;
}
