static int has_dotdot(const char *url) {
    /* Decode the URL and check for ".." path segments */
    size_t len = strlen(url);
    char *decoded = malloc(len + 1);
    if (!decoded) return 1; /* fail safe */
    size_t j = 0;
    for (size_t i = 0; i < len; i++) {
        if (url[i] == '%' && i + 2 < len) {
            int hi = hex_digit(url[i+1]);
            int lo = hex_digit(url[i+2]);
            if (hi >= 0 && lo >= 0) {
                decoded[j++] = (char)((hi << 4) | lo);
                i += 2;
                continue;
            }
        }
        decoded[j++] = url[i];
    }
    decoded[j] = '\0';
    int result = (strstr(decoded, "..") != NULL) ? 1 : 0;
    free(decoded);
    return result;
}
