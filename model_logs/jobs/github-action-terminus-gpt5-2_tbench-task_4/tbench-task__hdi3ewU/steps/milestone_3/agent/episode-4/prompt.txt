async function apiFetch(method, url, body, includeKey) {
  var opts = { method: method, headers: {} };

  // Only send API key + JSON Content-Type on POST/PUT/DELETE (not GET)
  if (includeKey) {
    opts.headers['Content-Type'] = 'application/json';
    opts.headers['X-API-Key'] = (API_KEY || '').trim();
    opts.body = JSON.stringify(body || {});
  }

  var res = await fetch(url, opts);
  return res;
}
