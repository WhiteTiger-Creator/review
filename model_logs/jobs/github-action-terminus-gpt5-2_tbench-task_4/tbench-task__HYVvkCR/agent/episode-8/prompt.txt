CREATE TRIGGER jobs_sync_type_insert AFTER INSERT ON jobs
BEGIN
  UPDATE jobs
     SET type=COALESCE(NEW.type, NEW.job_type),
         job_type=COALESCE(NEW.job_type, NEW.type)
   WHERE id=NEW.id;
END;
CREATE TRIGGER jobs_sync_type_update AFTER UPDATE OF type, job_type ON jobs
BEGIN
  UPDATE jobs
     SET type=COALESCE(NEW.type, NEW.job_type),
         job_type=COALESCE(NEW.job_type, NEW.type)
   WHERE id=NEW.id;
END;
