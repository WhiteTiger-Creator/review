func (p *WorkerPool) claimNextPending() (*pendingJob, error) {
        tx, err := p.db.Begin()
        if err != nil {
                return nil, err
        }
        defer tx.Rollback() //nolint:errcheck

        var j pendingJob
        err = tx.QueryRow(`
                SELECT id, type FROM jobs
                WHERE status = 'pending'
                ORDER BY priority DESC, created_at ASC
                LIMIT 1
        `).Scan(&j.id, &j.jobType)
        if err == sql.ErrNoRows {
                return nil, nil
        }
        if err != nil {
                return nil, err
        }

        res, err := tx.Exec(
                "UPDATE jobs SET status='running', updated_at=? WHERE id=? AND status='pending'",
                time.Now(), j.id,
        )
        if err != nil {
                return nil, err
        }
        n, _ := res.RowsAffected()
        if n == 0 {
                return nil, nil
        }
        return &j, tx.Commit()
}
