        if err != nil {
                return nil, err
        }
        defer tx.Rollback() //nolint:errcheck

        var j pendingJob
        err = tx.QueryRow(`
                SELECT id, type FROM jobs
                WHERE status = 'pending'
                ORDER BY priority DESC, created_at ASC
                LIMIT 1
        `).Scan(&j.id, &j.jobType)
        if err == sql.ErrNoRows {
                return nil, nil
        }

        }

                time.Now(), j.id,
        ); err != nil {

        }
        return &j, tx.Commit()
}
