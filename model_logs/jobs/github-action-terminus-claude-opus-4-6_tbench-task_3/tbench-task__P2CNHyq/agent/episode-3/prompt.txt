        w.Header().Set("Content-Type", "application/json")

}

}

func (s *Service) RegisterWorker(name string) (*Worker, error) {

        if err != nil {

                var count int
                if err := rows.Scan(&status, &count); err != nil {

                }
                switch status {
                case "pending":
