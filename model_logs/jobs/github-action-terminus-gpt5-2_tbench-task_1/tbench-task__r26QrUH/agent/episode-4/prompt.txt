static tb_status append_rec(FILE *f, tb_stamp s) {
  if (!fs_write_u32(f, s.gen))
    return TB_ERR_IO;

  // stamp format: u32 generation + u64 cursor (12 bytes)

  uint32_t lo = (uint32_t)(s.cursor & 0xffffffffu);
  uint32_t hi = (uint32_t)((s.cursor >> 32) & 0xffffffffu);
  if (!fs_write_u32(f, lo))
    return TB_ERR_IO;
  if (!fs_write_u32(f, hi))
    return TB_ERR_IO;

  return TB_OK;
}

static bool read_rec(FILE *f, tb_stamp *out) {
  uint32_t gen = 0;
  if (!fs_read_u32(f, &gen))
    return false;

  uint32_t lo = 0;
  if (!fs_read_u32(f, &lo))
    return false;

  uint32_t hi = 0;
  if (!fs_read_u32(f, &hi))
    return false;

  out->gen = gen;
  out->cursor = ((uint64_t)hi << 32) | (uint64_t)lo;
  return true;
}

tb_status journal_load_last(const char *path, tb_stamp *out, bool *found) {
  if (!out || !found)
    return TB_ERR_BAD_ARG;

  FILE *f = fopen(path, "rb");
  if (!f) {
    if (errno == ENOENT) {
      *found = false;
      return TB_OK;
    }
    return TB_ERR_IO;
  }

  tb_stamp cur = {0};
  bool any = false;
  while (read_rec(f, &cur)) {
    any = true;
  }
  fclose(f);
  if (!any) {
    *found = false;
    return TB_OK;
  }
  *found = true;
  *out = cur;
  return TB_OK;
}
