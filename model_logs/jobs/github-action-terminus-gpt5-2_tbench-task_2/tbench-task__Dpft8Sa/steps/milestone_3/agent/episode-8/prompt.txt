    if (!f) {

    int n = snprintf(hdr, sizeof(hdr),
                     "HTTP/1.1 200 OK\r\n"
                     "Connection: close\r\n"
                     "Content-Type: %s\r\n"
                     "Content-Length: %zu\r\n"
                     "\r\n",
                     ct, (size_t)st.st_size);
    if (n > 0) write_all(fd, hdr, (size_t)n);

    char buf[4096];
    size_t r;
    while ((r = fread(buf, 1, sizeof(buf), f)) > 0) {
        if (write_all(fd, buf, r) != 0) break;
    }
    fclose(f);
}
