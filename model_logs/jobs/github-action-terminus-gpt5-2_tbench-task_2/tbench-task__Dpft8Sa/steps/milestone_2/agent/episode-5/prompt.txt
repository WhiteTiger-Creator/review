This document is the normative REST contract for the notes API. Automated checks use arbitrary title and content strings in requests; only the behaviors, status
 codes, headers, and JSON field names here matter.

Store notes in /app/data/notes.db. Read the shared secret from /app/config/api.key and trim surrounding whitespace before comparing it to the X-API-Key header o
n mutating requests. The stock api.key file ships with tb3_notes_api_secret_2026. POST, PUT, and DELETE require a matching key or the server responds unauthoriz
ed. GET never requires the key; a bad or missing key on a mutating call is unauthorized.

Every JSON response must use a JSON content type and include an access-control header that allows any origin. OPTIONS preflight must advertise GET, POST, PUT, D
ELETE, and OPTIONS, and must allow Content-Type and X-API-Key as request headers. Preflight may return empty or non-empty success.

POST /api/notes accepts JSON with string title and content. Title must be present and non-empty; an empty object or missing or blank title is a client error. Su
ccess is HTTP 201 with a body containing numeric id plus stored title and content. GET /api/notes returns a JSON array of objects with id, title, and content. O
ptional limit and offset query parameters are integers: default limit 50, maximum 100, default offset 0. /api/notes?limit=100 returns at most one hundred rows;
/api/notes?limit=999 is capped the same way. The total row count before paging goes in the X-Total-Count header.

GET /api/notes/{id} returns that object. A missing id returns HTTP 404 with JSON {"error":"not found"} using that exact lowercase message string, including for
ids such as /api/notes/99999 on GET, PUT, or DELETE. PUT uses the same field rules as POST, returns HTTP 200 with saved fields when the row exists, or the same
not-found JSON when it does not. Successful DELETE returns HTTP 200 with JSON {"status":"deleted"}; deleting a missing id returns the same not-found JSON as GET
. Reject POST or PUT bodies larger than 65536 bytes with HTTP 413 and a JSON error body.

SQLite access must remain correct under concurrent requests. Verifiers may reset /app/data/notes.db and send bursts of about fifteen parallel POST requests; eve
ry stored row must still appear in responses afterward.

static enum MHD_Result send_json(struct MHD_Connection *connection,
                                  const char *body, unsigned int status_code) {
    struct MHD_Response *response = MHD_create_response_from_buffer(
        strlen(body), (void *)body, MHD_RESPMEM_MUST_COPY);
    MHD_add_response_header(response, "Content-Type", "application/json");
    MHD_add_response_header(response, "Access-Control-Allow-Origin", "*");
    MHD_add_response_header(response, "Access-Control-Allow-Methods",
                            "GET, POST, PUT, DELETE, OPTIONS");
    MHD_add_response_header(response, "Access-Control-Allow-Headers",
                            "Content-Type, X-API-Key");
    enum MHD_Result ret = MHD_queue_response(connection, status_code, response);
    MHD_destroy_response(response);
    return ret;
}

static enum MHD_Result send_json_total(struct MHD_Connection *connection,
                                       const char *body, unsigned int status_code,
                                       int total) {
    struct MHD_Response *response = MHD_create_response_from_buffer(
        strlen(body), (void *)body, MHD_RESPMEM_MUST_COPY);
    MHD_add_response_header(response, "Content-Type", "application/json");
    MHD_add_response_header(response, "Access-Control-Allow-Origin", "*");
    MHD_add_response_header(response, "Access-Control-Allow-Methods",
                            "GET, POST, PUT, DELETE, OPTIONS");
    MHD_add_response_header(response, "Access-Control-Allow-Headers",
                            "Content-Type, X-API-Key");
    char count_str[32];
    snprintf(count_str, sizeof(count_str), "%d", total);
    MHD_add_response_header(response, "X-Total-Count", count_str);
    enum MHD_Result ret = MHD_queue_response(connection, status_code, response);
    MHD_destroy_response(response);
    return ret;
}

static void request_completed(void *cls, struct MHD_Connection *connection,
                              void **con_cls, enum MHD_RequestTerminationCode toe) {
    (void)cls;
    (void)connection;
    (void)toe;
    if (con_cls && *con_cls) {
        struct connection_info *ci = *con_cls;
        if (ci->data)
            free(ci->data);
        free(ci);
        *con_cls = NULL;
    }
}

enum MHD_Result route_request(struct MHD_Connection *connection, const char *url,
                              const char *method, const char *upload_data,
                              size_t *upload_data_size, void **con_cls) {
    (void)upload_data;
    (void)upload_data_size;
    (void)con_cls;

    if (strcmp(method, "OPTIONS") == 0)
        return send_json(connection, "", MHD_HTTP_OK);

    if (strcmp(url, "/api/notes") == 0 && strcmp(method, "GET") == 0) {
        int total = get_total_notes_count();
        char *json = get_notes_paginated_json(50, 0);
        if (!json)
            return send_json_total(connection, "[]", MHD_HTTP_OK, total);
        enum MHD_Result r = send_json_total(connection, json, MHD_HTTP_OK, total);
        free(json);
        return r;
    }

    return send_json(connection, "{\"error\":\"not found\"}", MHD_HTTP_NOT_FOUND);
}
